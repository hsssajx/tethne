[docs]def new_meta_dict():
    """
    Creates a Meta Dictionary of citation data values. This function has the 
    most common values from citation data sources like WebOfScience, PubMed etc.
    
    meta_dict keys:
        * aulast -- authors' last name as a list.
        * auinit -- authors' first initial as a list.
        * atitle -- article title.
        * jtitle -- journal title or abbreviated title.
        * volume -- journal volume number.
        * issue -- journal issue number.
        * spage -- starting page of article in journal.
        * epage -- ending page of article in journal.
        * date -- article date of publication.
        * citations -- a list of meta_dict dictionaries for cited references.
        * ayjid -- First name (last, fi), publication year, and journal.
        * doi -- Digital Object Identifier.
        * pmid -- PubMed ID.
        * wosid -- Web of Science UT fieldtag.
    
    Returns
    -------
    meta_dict : dict
        A meta_list dictionary with 'None' as default values for all keys.
    
        
    """
    meta_dict = {'aulast':None,
                 'auinit':None,
                 'address':None, #added
                 'atitle':None,
                 'jtitle':None,
                 'volume':None,
                 'issue':None,
                 'spage':None,
                 'epage':None,
                 'date':None,
                 'citations':None, 
                 'addr1':None, #added
                 'country':None, #added
                 'addr2':None, #added
                 'addr3':None,#added
                 'ayjid':None,
                 'ainstid':None,#added author affiliation ID
                 'doi':None,
                 'pmid':None,
                 'wosid':None}
    
    return meta_dict

    """

                  'IS':'issue',

    return translator
